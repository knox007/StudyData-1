/// scanf 와 포인터 ////////////////////////////////////////////////////

-------------------------------
1.	scanf 함수 호출시 &연산자를 사용하는 이유
-------------------------------

	1.	
		int num;

		scanf("%d", &num);

		.....

		
		-	scanf 함수 호출이 완료되면 변수 num에는 값이 채워진다.

			scanf는 프로그램 사용자로부터 값을 입력 받아서 변수 num에 그 값을 채운다.


		-	scanf 외부의 변수에 값을 채우기 위해 변수의 주소값을 참조하여 접근한다.


	2.
		char str[30];

		scanf("%s", str);

		.....

		
		-	배열의 이름은 해당 배열의 주소를 나타내므로 &를 사용하지 않는다.
